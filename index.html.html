📋 SQL para criar a tabela no Supabase
Cole isso no SQL Editor do seu projeto Supabase e clique em Run:
CREATE TABLE IF NOT EXISTS sessoes (
  id         bigserial PRIMARY KEY,
  data       date NOT NULL,
  materia    text NOT NULL,
  topico     text NOT NULL,
  feitas     int  NOT NULL DEFAULT 0,
  acertos    int  NOT NULL DEFAULT 0,
  tempo      int  NOT NULL DEFAULT 0,
  tipo       text NOT NULL DEFAULT 'questoes',
  revisado   boolean DEFAULT false,
  created_at timestamptz DEFAULT now()
);

-- Habilitar Row Level Security (leitura pública para demo)
ALTER TABLE sessoes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON sessoes FOR ALL USING (true) WITH CHECK (true);
